## Stack using single queue


```swift
struct StackQueue<T> {
    let size: Int
    var queue: CircularQueue<T>
    
    init(size: Int) {
        self.size = size
        self.queue = CircularQueue<T>(size: size)
    }
    

    var top: T? {
        queue.peek
    }

    mutating func push(_ element: T) {
        if !(queue.enqueue(element)) {
            print("queue is full")
            return
        }
        
        var tempCounter = queue.count - 1
        while(tempCounter > 0) {
            if let ele = queue.dequeue() {
                queue.enqueue(ele)
            }
            tempCounter -= 1
        }
    }

    @discardableResult
    mutating func pop() -> T? {
        queue.dequeue()
    }

    func data() {
        print("elements \(queue)")
    }
}

```

Time Complexity:
push - O(N) and pop - O(1)
