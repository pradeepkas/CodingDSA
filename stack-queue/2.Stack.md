```swift
struct StackArray<T> {
    private var elements: [T] = []

    var top: T? {
        elements.last
    }

    mutating func push(_ element: T) {
        elements.append(element)
    }

    @discardableResult
    mutating func pop() -> T? {
        elements.removeLast()
    }

    func data() {
        print("elements \(elements)")
    }
}
```
