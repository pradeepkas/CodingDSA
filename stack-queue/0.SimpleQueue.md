
## Simple Queue

> Rule: FIFO or LILO

```swift
struct Queue<T> {
    var elements: [T?] = []

    let size: Int
    init(size: Int) {
        self.size = size
        elements = Array(repeating: nil, count: size)
    }

    private var front = 0
    private var rear = 0
    private var count = 0

    mutating func enqueue(_ ele: T) -> Bool {
        if count == size || rear >= size { // queue full
            //rear >= size  this is requried in case after full quueue you did dequeu in
            //this count will 3 but rear will be 4 menas we can enter any other element.
            return false
        }
        elements[rear] = ele
        rear += 1
        count += 1
        return true
    }

    mutating func dequeue() -> T? {
        if count == 0 {
            return nil
        }

        let temp = elements[front]
        elements[front] = nil
        front += 1
        count -= 1
        return temp
    }

    var peek: T? {
        elements[front]
    }
}

```
