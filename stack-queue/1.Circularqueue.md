
## CircularQueue


```swift
struct CircularQueue<T> {
    var elements: [T?] = []

    let size: Int
    init(size: Int) {
        self.size = size
        elements = Array(repeating: nil, count: size)
    }

    private var front = 0
    private var rear = 0
    var count = 0

    mutating func enqueue(_ ele: T) -> Bool {
        if count == size {
            return false
        }
        elements[rear] = ele
        rear = (rear + 1) % size
        count += 1
        return true
    }

    mutating func dequeue() -> T? {
        if count == 0 {
            return nil
        }

        let temp = elements[front]
        elements[front] = nil
        front = (front + 1) % size
        count -= 1
        return temp
    }

    var peek: T? {
        elements[front]
    }
}
```
