## asteroid-collision

https://leetcode.com/problems/asteroid-collision/description/


```swift
    func asteroidCollision(_ asteroids: [Int]) -> [Int] {
        var stack = StackArray<Int>()
        
        for element in asteroids {
           if element > 0 {
                stack.push(element)
            } else {
                while stack.count >= 1 && stack.top! > 0 && stack.top! < abs(element) {
                    stack.pop()
                }
                
                if stack.count == 0 {
                    stack.push(element)
                }
                else if stack.top! == abs(element) {
                    stack.pop() // if elemnets as same will destroye each other
                } else if stack.top! < abs(element) { // it will allow negtive as well in this stack 
                    stack.push(element)
                }
            }
        }
        return stack.elements
    }
```
