## 141. Linked List Cycle

## https://leetcode.com/problems/linked-list-cycle/description/


```swift
    func isCycleExist(_ head: Node?) -> Bool {
        var slow: Node? = head
        var fast: Node? = head
        
        while fast != nil && fast?.next != nil {
            fast = fast?.next?.next
            slow = slow?.next
            if fast === slow {
                return true
            }
        }
        return false
    }
```
