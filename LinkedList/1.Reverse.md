```swift
    func reverseList(_ head: ListNode?) -> ListNode? {
        var temp: ListNode? = head
        var newLinked: ListNode?
        
        while (temp != nil ) {
            if newLinked == nil {
                newLinked = ListNode(temp?.val ?? 0)
            } else {
                let node = ListNode(temp?.val ?? 0)
                node.next = newLinked
                newLinked = node
            }
            temp = temp?.next
        }
        return newLinked

    }
```
